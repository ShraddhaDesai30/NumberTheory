#include <bits/stdc++.h>
using namespace std;

void sieve() {
    int maxN = 1000000;
    bool is_prime[maxN + 1];

    for (int i = 0; i <= maxN; i++) {
        is_prime[i] = true;
    }

    is_prime[0] = is_prime[1] = false;

    for (int i = 2; i <= maxN; i++) {
        if (is_prime[i]) {
            for (int j = 2 * i; j <= maxN; j += i) {
                is_prime[j] = false;
            }
        }
    }

    for (int i = 2; i <= 30; i++) {
        if (is_prime[i]) {
            cout << i << " ";
        }
    }
}

int main() {
    sieve();

    return 0;
}
